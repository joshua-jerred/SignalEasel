49 void addSymbol(Phase phase, bool filter_end_of_symbol) {
50 current_phase_ = phase;
51 filter_end_of_symbol_ = filter_end_of_symbol;
52
53 constexpr std::array<double, 4> SHIFT_ANGLE = {0.0, bst::HALF_PI, bst::PI,
54 bst::THREE_PI_HALVES};
55
56 for (sample_num_ = 0; sample_num_ < SAMPLES_PER_SYMBOL_; sample_num_++) {
57 // Calculate X component of the carrier wave, from -1.0 to 1.0
58 double raw_sample = std::sin(carrier_wave_angle_ +
59 SHIFT_ANGLE[static_cast<uint8_t>(phase)]);
60
61 // Apply the envelope filter to the sample
62 envelopeFilter(raw_sample);
63
64 // Convert the wave position to a int16_t sample
65 uint16_t sample =
66 static_cast<int16_t>((raw_sample * signal_easel::MAX_SAMPLE_VALUE));
67
68 sample_buffer_.push_back(sample);
69
70 carrier_wave_angle_ += angle_delta_;
71 if (carrier_wave_angle_ >= bst::TWO_PI) {
72 carrier_wave_angle_ -= bst::TWO_PI;
73 }
74 }
75
76 previous_phase_ = current_phase_;
77 }

86 void envelopeFilter(double &raw_sample) const {
87 // Determine if the current sample is in the first half of the symbol
88 const bool first_half = sample_num_ < HALF_SAMPLES_PER_SYMBOL_;
89
90 // If the current phase is different from the previous phase, apply the
91 // filter to the first half of the symbol. If the next phase is different
92 // from the current, apply the filter to the second half of the symbol.
93 const bool filter_first_half =
94 first_half && current_phase_ != previous_phase_;
95
96 // Apply the filter to the sample
97 if ((first_half && filter_first_half) ||
98 (!first_half && filter_end_of_symbol_)) {
99 raw_sample *=
100 std::sin(FILTER_SIN_PI_SCALER_ * static_cast<double>(sample_num_));
101 }
102
103 // Apply the amplitude scaling to the sample
104 raw_sample *= AMPLITUDE_;
105 }
